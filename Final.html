def MobiusComp(z,zs,zd,m):
  #zd is the drain coord, zs is the source coord, m is how much you step towards the drain.
  #m=1 gives zd, m=-1 gives zs, and imaginary values give rotations
  return (2*m*zd*zs-z*((m+1)*zd+(m-1)*zs))/(zs-zd+m*(zd+zs-2*z))
def Mobius2DReal(p,ps,pd,pm):
  #zd is the drain coord, zs is the source coord, m is how much you step towards the drain.
  #m=1 gives zd, m=-1 gives zs, and imaginary values give rotations
  z=MobiusComp(complex(*p),complex(*ps),complex(*pd),complex(*pm))
  return (z.real,z.imag)

from turtle_setup import *
# Function to draw a fractal tree using recursion
def fractal_tree(length, depth):
    if depth == 0:
        return
    else:

        forward(length)
        left(30)

        fractal_tree(length * 0.7, depth - 1)

        right(60)

        fractal_tree(length * 0.7, depth - 1)

        left(30)
        backward(length)
def calGraphElement(length, depth,ang,angMult,lengthMult):
  if depth == 0:
    return
  else:
    forward(length)
    calGraphElement(length*lengthMult, depth-1,ang*angMult,angMult,lengthMult)
    left(ang)
    calGraphElement(length*lengthMult, depth-1,ang*angMult,angMult,lengthMult)
    right(2*ang)
    calGraphElement(length*lengthMult, depth-1,ang*angMult,angMult,lengthMult)
    left(ang)
    backward(length)
def calGraph(length, depth,ang,angMult,lengthMult):
  calGraphElement(length, depth,ang,angMult,lengthMult)
  right(90)
  calGraphElement(length, depth,ang,angMult,lengthMult)
  right(90)
  calGraphElement(length, depth,ang,angMult,lengthMult)
  right(90)
  calGraphElement(length, depth,ang,angMult,lengthMult)
  right(90)
# Initialize the turtle
initializeTurtle(initial_window_size=(500, 500))

# Set up the turtle's starting position and orientation
jump(250, 250)  # Starting position at the bottom center of the canvas
face(0)
color('green')  # Set the pen color

# Set the length and depth for the fractal tree
length = 100  # Initial length of the main trunk
depth = 7     # Recursion depth (number of branching levels)

# Draw the fractal tree
calGraph(length, depth,90,1,0.49)

# Display the drawing
show()
